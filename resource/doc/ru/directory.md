# Структура каталогов
```
.
├── app                           Директория приложения
│   ├── controller                Директория контроллеров
│   ├── model                     Директория моделей
│   ├── view                      Директория представлений
│   ├── middleware                Директория промежуточного ПО
│   │   └── StaticFile.php        Встроенное промежуточное ПО для статических файлов
|   └── functions.php             Здесь пишутся пользовательские бизнес-функции
|
├── config                        Директория конфигурации
│   ├── app.php                   Конфигурация приложения
│   ├── autoload.php              Файлы, указанные здесь, будут автоматически загружены
│   ├── bootstrap.php             Конфигурация обратного вызова для запуска процесса onWorkerStart
│   ├── container.php             Конфигурация контейнера
│   ├── dependence.php            Конфигурация зависимостей контейнера
│   ├── database.php              Конфигурация базы данных
│   ├── exception.php             Конфигурация исключений
│   ├── log.php                   Конфигурация журнала
│   ├── middleware.php            Конфигурация промежуточного ПО
│   ├── process.php               Конфигурация пользовательских процессов
│   ├── redis.php                 Конфигурация redis
│   ├── route.php                 Конфигурация маршрутов
│   ├── server.php                Конфигурация сервера порта, количества процессов и т. д.
│   ├── view.php                  Конфигурация представлений
│   ├── static.php                Конфигурация переключателя статических файлов и промежуточного ПО для статических файлов
│   ├── translation.php           Конфигурация многоязычности
│   └── session.php               Конфигурация сессий
├── public                        Директория статических ресурсов
├── process                       Директория пользовательских процессов
├── runtime                       Директория времени выполнения приложения, требующая прав на запись
├── start.php                     Файл запуска службы
├── vendor                        Директория установленных с помощью composer сторонних библиотек
└── support                       Адаптация библиотек (включая библиотеки сторонних разработчиков)
    ├── Request.php               Класс запроса
    ├── Response.php              Класс ответа
    ├── Plugin.php                Скрипт установки и удаления плагинов
    ├── helpers.php               Вспомогательные функции (для пользовательских бизнес-функций используйте app/functions.php)
    └── bootstrap.php             Скрипт инициализации после запуска процесса
```
